char* longestCommonPrefix(char** strs, int strsSize) {
    char* res=strdup(strs[0]);
    res[strlen(res)]='\0';
    for(int i=1;i<strsSize;i++)
    {
        int j=0;
        while(res[j]!='\0'&&strs[i][j]!='\0'&&res[j]==strs[i][j])
        {
            j++;
        }
        if(res[j]!=strs[i][j])
        {
            res[j]='\0';
        }
       
    }
    return res;
   
}
